#include <vector>

int main(int, char **) {

  int iArray[]{1, 2, 3, 4, 5, 6, 7, 8, 9, 0};

  // access the elements by value
  for (int i : iArray) {
    // do something
  };

  std::vector<int> v{1, 2, 3, 4};

  // access the elements as references
  for (const auto &i : v) {
    // do something
  }
}
